using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace Методи_вводу
{
    class Program
    {
        static void Main(string[] args)
        {
            double x = double.Parse(Console.ReadLine());   //перший варіант  
            double y = Convert.ToDouble(Console.ReadLine()); //Другий вар. вибирати 1.   

            Console.WriteLine("sin({0}) = {1}",x,Math.Sin(x)); // синус числа
            Console.WriteLine("{0}^2 = {1}", y, Math.Pow(y, 2));  // до квадрату
            Console.WriteLine("square root of {0} = {1}", x, Math.Sqrt(x));  //корінь числа
            Console.WriteLine("minimym = {0}",Math.Min(x, y));  //мінімальне значення
            Console.WriteLine("minimym = {0}", Math.Max(x, y));  //maxмальне значення

        }
    }
}
