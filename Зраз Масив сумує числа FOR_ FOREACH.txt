using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleApp16
{
    class Program
    {
        static void Main(string[] args)
        {
            int[] mas = { 11, 2, 33, 4, 5 };//пишемо масив не вводимо
            int sum = 0;//вводимо змінну
            for (int i = 0; i < mas.Length; i++)//умова циклу
            {
                sum += mas[i];//рахує суму всіх єлементів
            }
            Console.Write($"Sum all number of array {sum}");
            Console.WriteLine();


            sum = 0;
            foreach (int x in mas)
            {
                sum += x;
            }
            Console.WriteLine($"Sum = {sum}");


        }





    }
}
